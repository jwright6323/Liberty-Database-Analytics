def test_init()
  t = Mysql::Time.new
  assert_equal(0, t.year);
  assert_equal(0, t.month);
  assert_equal(0, t.day);
  assert_equal(0, t.hour);
  assert_equal(0, t.minute);
  assert_equal(0, t.second);
  assert_equal(false, t.neg);
  assert_equal(0, t.second_part);
end
