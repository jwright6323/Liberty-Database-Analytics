def to_gplot (io = "")
  @sets.each { |var, val| io << "set #{var} #{val}\n" }

  if @data.size > 0 then
    io << @cmd << " " 
    io << @data.collect { |e| e.plot_args }.join(", ")
    io << "\n"

    @data.each do |ds| 
      io << ds.to_gsplot << "e\n"
    end
  end

  io
end
